function speak(text) { const speech = new SpeechSynthesisUtterance(text); function loadVoiceAndSpeak() { let voices = speechSynthesis.getVoices(); console.log(voices); // برای تست let faVoice = voices.find(v => v.lang === "fa-IR"); if (faVoice) { speech.voice = faVoice; speech.lang = "fa-IR"; } else { let enVoice = voices.find(v => v.lang.includes("en")); speech.voice = enVoice; speech.lang = "en-US"; } speech.rate = 0.9; speech.pitch = 1; speechSynthesis.cancel(); speechSynthesis.speak(speech); } // اگر هنوز صداها آماده نبودن if (speechSynthesis.getVoices().length === 0) { speechSynthesis.onvoiceschanged = loadVoiceAndSpeak; } else { loadVoiceAndSpeak(); } }
