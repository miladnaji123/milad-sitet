function speak(text) { const speech = new SpeechSynthesisUtterance(text); let voices = window.speechSynthesis.getVoices(); // لیست صداها رو چاپ کن (برای تست) console.log(voices); // اول فارسی let faVoice = voices.find(v => v.lang.includes("fa")); // اگر فارسی نبود، انگلیسی خوب انتخاب کن let enVoice = voices.find(v => v.lang.includes("en")); speech.voice = faVoice || enVoice; speech.lang = faVoice ? "fa-IR" : "en-US"; speech.rate = 0.9; speech.pitch = 1; window.speechSynthesis.cancel(); window.speechSynthesis.speak(speech); }
