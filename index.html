function speak(text) { const speech = new SpeechSynthesisUtterance(text); function loadVoiceAndSpeak() { let voices = speechSynthesis.getVoices(); let faVoice = voices.find(v => v.lang.includes("fa")); if (faVoice) { speech.voice = faVoice; speech.lang = "fa-IR"; } else { speech.lang = "en-US"; } speech.rate = 0.9; speech.pitch = 1; speechSynthesis.cancel(); speechSynthesis.speak(speech); } if (speechSynthesis.getVoices().length === 0) { speechSynthesis.onvoiceschanged = loadVoiceAndSpeak; } else { loadVoiceAndSpeak(); } }
